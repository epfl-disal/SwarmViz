1function y = sammon(x)
  2%#codgen
  3%
  4% SAMMON - apply Sammon's nonlinear mapping 
  5%
  6%    Y = SAMMON(X) applies Sammon's nonlinear mapping procedure on
  7%    multivariate data X, where each row represents a pattern and each column
  8%    represents a feature.  On completion, Y contains the corresponding
  9%    co-ordinates of each point on the map.  By default, a two-dimensional
 10%    map is created.  Note if X contains any duplicated rows, SAMMON will
 11%    fail (ungracefully). 
 12%
 13%    [Y,E] = SAMMON(X) also returns the value of the cost function in E (i.e.
 14%    the stress of the mapping).
 15%
 16%    An N-dimensional output map is generated by Y = SAMMON(X,N) .
 17%
 18%    A set of optimisation options can also be specified using a third
 19%    argument, Y = SAMMON(X,N,OPTS) , where OPTS is a structure with fields:
 20%
 21%       MaxIter        - maximum number of iterations
 22%       TolFun         - relative tolerance on objective function
 23%       MaxHalves      - maximum number of step halvings
 24%       Input          - {'raw','distance'} if set to 'distance', X is 
 25%                        interpreted as a matrix of pairwise distances.
 26%       Display        - {'off', 'on', 'iter'}
 27%       Initialisation - {'pca', 'random'}
 28%
 29%    The default options structure can be retrieved by calling SAMMON with
 30%    no parameters.
 31%
 32%    References :
 33%
 34%       [1] Sammon, John W. Jr., "A Nonlinear Mapping for Data Structure
 35%           Analysis", IEEE Transactions on Computers, vol. C-18, no. 5,
 36%           pp 401-409, May 1969.
 37%
 38%    See also : SAMMON_TEST
 39
 40%
 41% File        : sammon.m
 42%
 43% Date        : Monday 12th November 2007.
 44%
 45% Author      : Gavin C. Cawley and Nicola L. C. Talbot
 46%
 47% Description : Simple vectorised MATLAB implementation of Sammon's non-linear
 48%               mapping algorithm [1].
 49%
 50% References  : [1] Sammon, John W. Jr., "A Nonlinear Mapping for Data
 51%                   Structure Analysis", IEEE Transactions on Computers,
 52%                   vol. C-18, no. 5, pp 401-409, May 1969.
 53%
 54% History     : 10/08/2004 - v1.00
 55%               11/08/2004 - v1.10 Hessian made positive semidefinite
 56%               13/08/2004 - v1.11 minor optimisation
 57%               12/11/2007 - v1.20 initialisation using the first n principal
 58%                                  components.
 59%
 60% Thanks      : Dr Nick Hamilton (nick@maths.uq.edu.au) for supplying the
 61%               code for implementing initialisation using the first n
 62%               principal components (introduced in v1.20).
 63%
 64% To do       : The current version does not take advantage of the symmetry
 65%               of the distance matrix in order to allow for easy
 66%               vectorisation.  This may not be a good choice for very large
 67%               datasets, so perhaps one day I'll get around to doing a MEX
 68%               version using the BLAS library etc. for very large datasets.
 69%

 71%

 73%    it under the terms of the GNU General Public License as published by
 74%    the Free Software Foundation; either version 2 of the License, or
 75%    (at your option) any later version.
 76%
 77%    This program is distributed in the hope that it will be useful,
 78%    but WITHOUT ANY WARRANTY; without even the implied warranty of
 79%    MERCHANTABILITY or FITNESS FOR A PARTICULAR PURPOSE.  See the
 80%    GNU General Public License for more details.
 81%
 82%    You should have received a copy of the GNU General Public License
 83%    along with this program; if not, write to the Free Software
 84%    Foundation, Inc., 59 Temple Place, Suite 330, Boston, MA 02111-1307 USA

 86
 87% use the default options structure
 88Initialisation = 'random';
 89
 90n = 2;
 91
 92% create distance matrix unless given by parameters
 93D = euclid(x, x);
 94
 95
 96% remaining initialisation
 97
 98N     = size(x, 1);
 99scale = 0.5/sum(D(:));
100D     = D + eye(N);
101Dinv  = 1./D;
102
103if strcmp(Initialisation, 'pca')
104   [UU,DD] = svd(x);
105   y       = UU(:,1:n)*DD(1:n,1:n);
106else
107   y = randn(N, n);
108end
109
110one   = ones(N,n);
111d     = euclid(y,y) + eye(N);
112dinv  = 1./d;
113delta = D - d;
114E     = sum(sum((delta.^2).*Dinv));
115
116% get on with it
117
118for i=1:500
119
120   % compute gradient, Hessian and search direction (note it is actually
121   % 1/4 of the gradient and Hessian, but the step size is just the ratio
122   % of the gradient and the diagonal of the Hessian so it doesn't matter).
123
124   delta    = dinv - Dinv;
125   deltaone = delta*one;
126   g        = delta*y - y.*deltaone;
127   dinv3    = dinv.^3; 
128   y2       = y.^2;
129   H        = dinv3*y2 - deltaone -2*y.*(dinv3*y) + y2.*(dinv3*one);
130   s        = -g(:)./abs(H(:));
131   y_old    = y;
132
133   % use step-halving procedure to ensure progress is made
134
135   for j=1:20
136
137      y(:) = y_old(:) + s;
138
139      d     = euclid(y,y) + eye(N);
140      dinv  = 1./d;
141      delta = D - d;
142      E_new = sum(sum((delta.^2).*Dinv));
143
144      if E_new < E
145
146         break;
147
148      else
149
150         s = 0.5*s;
151
152      end
153
154   end
155
156   % bomb out if too many halving steps are required
157
158   if j == 20
159      break;
160   end
161
162   % evaluate termination criterion
163
164   if abs((E - E_new)/E) < 1E-9
165      break;
166   end
167
168   % report progress
169   E = E_new;
170
171end
172
173% fiddle stress to match the original Sammon paper
174
175E = E*scale;
176
177% all done
178
